#include <TimerOne.h>
int i=0;
const byte  seg[10]=
{
   B1111110,
   B0110000,
   B1101101,
   B1111001,
   B0110011,
   B1011011,
   B1011111,
   B1110000,
   B1111111,
   B1111011  
};
void setup() 
{
      DDRA = B1111111;
      DDRC = B1111;
      Timer1.initialize(1000000);
      Timer1.attachInterrupt(callback);   
    //  PORTC= 1101;
}
 
void loop()
{
      if(i>9)
      {
          i=0;  
      }
      else
      {
          PORTA = seg[i];
      }
}
void callback()                        
{                        
    i++;
}
